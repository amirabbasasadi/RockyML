65template<typename T_e, int T_dim>
66class sync_broadcast_best: public mpi_strategy<T_e, T_dim>{
67protected:
68 basic_scontainer<T_e, T_dim>* cluster_best_container_;
69public:
70 void set_target_container(basic_scontainer<T_e, T_dim>* container){
71 this->set_target_container = container;
72 }
73 sync_broadcast_best(basic_scontainer<T_e, T_dim>* container){
74 this->cluster_best_container_ = container;
75 this->fetch_mpi_info();
76 }
77 sync_broadcast_best(){
78 this->cluster_best_container_ = nullptr;
79 this->fetch_mpi_info();
80 }
81 virtual void apply(){
82 // identify the process
83 int rank = this->mpi_rank();
84 // ask everyone in the cluster to find the min value
85 struct {
86 T_e cluster_best_min;
87 int rank;
88 } data_out, result;
89
90 data_out.cluster_best_min = cluster_best_container_->values[0];
91 data_out.rank = rank;
92
93 if constexpr(std::is_same<T_e, double>::value){
94 MPI_Allreduce(&data_out, &result, 1, MPI_DOUBLE_INT, MPI_MINLOC, MPI_COMM_WORLD);
95 MPI_Bcast(cluster_best_container_->particle(0), T_dim, MPI_DOUBLE, result.rank, MPI_COMM_WORLD);
96 }
97 if constexpr(std::is_same<T_e, float>::value){
98 MPI_Allreduce(&data_out, &result, 1, MPI_FLOAT_INT, MPI_MINLOC, MPI_COMM_WORLD);
99 MPI_Bcast(cluster_best_container_->particle(0), T_dim, MPI_FLOAT, result.rank, MPI_COMM_WORLD);
100 }
101 cluster_best_container_->values[0] = result.cluster_best_min;
102 }
103};
