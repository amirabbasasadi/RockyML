54  loss_projection_2d(system<T_e>* problem, std::string label, int width, int height, T_e x_min, T_e y_min, T_e x_max, T_e y_max){
55  problem_ = problem;
56  label_ = label;
57  width_ = width;
58  height_ = height;
59  x_min_ = x_min;
60  y_min_ = y_min;
61  x_max_ = x_max;
62  y_max_ = y_max;
63  step_ = 0;
64  // obtain resolution settings
65  delta_x_ = (x_max_ - x_min_) / static_cast<double>(width_);
66  delta_y_ = (y_max_ - y_min_) / static_cast<double>(height_);
67  // real image size after possible numerical errors
68  l_width_ = (x_max_ - x_min_) / delta_x_ ;
69  l_height_ = (y_max_ - y_min_) / delta_y_;
70  // allocate memory for the mesh
71  z.resize(l_height_);
72  for(int y=0; y<l_height_; y++)
73  z[y].resize(l_width_);
74  }
75  void save_mesh(){
76  std::string path = fmt::format("zagros_{}_{}.data", label_, step_);
77  std::fstream handler(path, std::fstream::out);
78  // writing metadata
79  handler << step_ << std::endl;
80  handler << fmt::format("{} {} {} {}", x_min_, y_min_, x_max_, y_max_) << std::endl;
81  handler << fmt::format("{} {}", l_height_, l_width_) << std::endl;
82  // writing the mesh values
83  for(int y=0; y<l_height_; y++)
84  for(int x=0; x<l_width_; x++)
85  handler << z[y][x] << " ";
86  handler.close();
87  }
88  virtual void apply(){
89  // traverse the 2D mesh in parallel
90  tbb::parallel_for(0, this->l_width_, [&](auto x){
91  tbb::parallel_for(0, this->l_height_, [&](auto y){
92  T_e thread_point[2];
93  thread_point[0] = x * this->delta_x_ + this->x_min_;
94  thread_point[1] = y * this->delta_y_ + this->y_min_;
95  this->z[y][x] = this->problem_->objective(thread_point);
96  });
97  });
98 
99  save_mesh();
100  step_++;
101  }
102 };

104 };
