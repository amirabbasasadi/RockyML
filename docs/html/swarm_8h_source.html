271  void update_cluster_best(){
272  // identify the process
273  int rank = this->mpi_rank();
274  // ask everyone in the cluster to find the min value
275  struct {
276  T_e cluster_best_min;
277  int rank;
278  } data_out, result;
279 
280  data_out.cluster_best_min = global_best_min_;
281  data_out.rank = rank;
282 
283  if constexpr(std::is_same<T_e, double>::value){
284  MPI_Allreduce(&data_out, &result, 1, MPI_DOUBLE_INT, MPI_MINLOC, MPI_COMM_WORLD);
285  }
286  if constexpr(std::is_same<T_e, float>::value){
287  MPI_Allreduce(&data_out, &result, 1, MPI_FLOAT_INT, MPI_MINLOC, MPI_COMM_WORLD);
288  }
289 
290  // the owner of best solution should broadcast its solution to the others
291  if(rank == result.rank){
292  std::copy(global_best_argmin_,
293  global_best_argmin_ + T_dim,
294  cluster_best_argmin_);
295  }
296  if constexpr(std::is_same<T_e, double>::value){
297  MPI_Bcast(cluster_best_argmin_, T_dim, MPI_DOUBLE, result.rank, MPI_COMM_WORLD);
298  }
299  if constexpr(std::is_same<T_e, float>::value){
300  MPI_Bcast(cluster_best_argmin_, T_dim, MPI_FLOAT, result.rank, MPI_COMM_WORLD);
301  }

303  }
310  std::pair<T_e, int> find_global_best_parallel(){

312  // initial answer
313  std::make_pair(std::numeric_limits<T_e>::max(), 0),
314  // reducing each chunk
315  [this](const tbb::blocked_range<T_e>& r, std::pair<T_e, int> init) -> std::pair<T_e, int>{
316  for (int i=r.begin(); i!=r.end(); i++){
317  if (this->particles_best_min_[i] < init.first)
318  init = std::make_pair(this->particles_best_min_[i], i);
319  }
320  return init;
321  },
322  // reducing the result of chunks

324  if(x.first < y.first)
325  return x;
326  return y;
327  });
