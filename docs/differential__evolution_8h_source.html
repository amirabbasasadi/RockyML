45  // apply differential evolution within groups in parallel
46  virtual void apply(){
47  tbb::parallel_for(0, n_crossovers_, [this](int p){
48  int parents[4];
49  this->container_->sample_n_particles(parents, 4);
50  auto& [x, a, b, c] = parents;
51  // making a copy of x
52  std::copy(this->container_->particle(x),
53  this->container_->particle(x) + T_dim,
54  this->candidates_->particle(p));
55  for(int d=0; d<T_dim; d++){
56  // perform crossover based on the crossover probability CR
57  if(this->rand_uniform() > this->CR_)
58  continue;
59  // apply crossover
60  this->candidates_->particles[p][d] = this->container_->particles[a][d] +
61  this->DW_ * (this->container_->particles[b][d] -
62  this->container_->particles[c][d]);
63  }
64  });
