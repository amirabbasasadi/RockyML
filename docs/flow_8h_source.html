28 template<typename T_e, int T_dim, int T_block_dim>
29 class runtime_storage{
30 public:
31  // allocated containers
32  std::vector<std::unique_ptr<basic_scontainer<T_e, T_block_dim>>> cnt_storage;
33  // allocated strategies
34  std::map<int, std::vector<std::unique_ptr<basic_strategy<T_e, T_block_dim>>>> str_storage;
35  // mapping the containers' id to their storage blocks
36  std::map<std::string, int> cnt_map;
37  // iteration counter for nodes who runs periodically
38  std::map<int, int> iter_counter;
39  // tracking the convergence of containers
40  std::map<int, std::deque<T_e>> conv_tracker;
41  // a mask representing active variables in blocked descent
42  std::vector<int> bcd_mask;
43  // state of blocked systems
44  std::unique_ptr<basic_scontainer<T_e, T_dim>> blocked_state;
45  tbb::enumerable_thread_specific<std::vector<T_e>> th_blocked_states;
46  // a temp buffer for broadcasting best partial solution
47  std::unique_ptr<basic_scontainer<T_e, T_block_dim>> partial_best;
48  // amount of allocated memory
49  size_t container_space(){
50  size_t allocated_mem = 0;
51  for(auto const& cnt: cnt_storage)
52  allocated_mem += cnt->space();
53  if constexpr(T_dim != T_block_dim){
54  allocated_mem += blocked_state->space();
55  allocated_mem += partial_best->space();
56  for(auto const& th_state: th_blocked_states)
57  allocated_mem += th_state.size() * sizeof(T_e);
58  }
59  return allocated_mem;
60  }
66  bool container_exist(std::string id){
67  auto it = cnt_map.find(id);
68  return it != cnt_map.end();
69  }
76  basic_scontainer<T_e, T_block_dim>* container(std::string id){
77  if (id == std::string("__best__"))
78  return partial_best.get();
79 
80  if (!container_exist(id)){
81  spdlog::warn("container {} does not exist", id);
82  }
83  // find the target container
84  int target_cnt_ind = cnt_map[id];
85  return cnt_storage[target_cnt_ind].get();
86  }
87  void allocate_container(std::string id, int n_particles, int group_size){
88  // allocate a continer
89  auto cnt = std::make_unique<basic_scontainer<T_e, T_block_dim>>(n_particles, group_size);
90  cnt->allocate();
91  spdlog::info("container {} was allocated. size : {:.2f} MB", id, cnt->space()/(1024.0*1024.0));
92  cnt_storage.push_back(std::move(cnt));
93  // register the id in the storage
94  cnt_map[id] = cnt_storage.size()-1;
95  }
96  // find the best partial solution in the storage
97  void update_partial_best(){
98  std::pair<T_e, int> best;
99  int best_ci = -1;
100  best.first = std::numeric_limits<T_e>::max();
101  for(int ci=0; ci<cnt_storage.size(); ci++){
102  auto min_and_index = cnt_storage[ci]->best_min_index();
103  if((min_and_index.first < best.first) || best_ci == -1){
104  best = min_and_index;
105  best_ci = ci;
106  }
107  }
108  if (best.first < partial_best->values[0]){
109  // copy the best solution to the container
110  std::copy(cnt_storage[best_ci]->particle(best.second),
111  cnt_storage[best_ci]->particle(best.second)+T_block_dim,
112  partial_best->particle(0));
113 
114  // copy the corresponding min value
115  partial_best->values[0] = best.first;
116  }
117  }
118  // synchronize best partial solution
119  void sync_partial_best(){
120  // Assumption : update_partial_best has been called already
121  // This function must be called before regenerating BCD mask
122  sync_broadcast_best<T_e, T_block_dim> sync_best_partial_str(partial_best.get());
123  sync_best_partial_str.apply();
124  // replace the old partial solution in solution states
125  for(int i=0; i<T_block_dim; i++)
126  blocked_state->particles[0][bcd_mask[i]] = partial_best->particles[0][i];
127  // replace the old partial solution in thread-specific states
128  int i=0;
129  for(auto& th_state: th_blocked_states){
130  th_state[bcd_mask[i]] = partial_best->particles[0][i];
131  i++;
132  }
133  }
134  // reset all solution containers
135  void reset(){
136  for(auto& cnt: cnt_storage)
137  cnt->reset_values();
138  for(auto& [tag, str_vec]: str_storage)
139  for(auto& str: str_vec)
140  str->reset();
141  }
142 };

400 template<typename T_e, int T_dim, int T_block_dim, typename T_traverse_fn>
401 struct running_visitor{
402  // objective system
403  system<T_e>* problem;
404  // visitor will change main storage
405  runtime_storage<T_e, T_dim, T_block_dim>* main_storage;
406  // visitor may also change the path stack in the case of composable flows
407  T_traverse_fn* traverse_fn;
408 
409  template<typename T_n>
410  void operator()(T_n node){
411  if constexpr (std::is_base_of<dena::run_node, T_n>::value){
412  // run n times
413  if constexpr (std::is_base_of<dena::run_n_times_node, T_n>::value){
414  for(int i=0; i<node.n_iters; i++){
415  (*traverse_fn)(node.sub_procedure.front(), problem, main_storage);
416  }
417  }
418  // run with a probability
419  if constexpr (std::is_base_of<dena::run_with_probability_node, T_n>::value){
420  float sample = utils::random::uniform<float>(0.0, 1.0);
421  if(sample < node.prob){
422  (*traverse_fn)(node.sub_procedure.front(), problem, main_storage);
423  }
424  }
425  if constexpr (std::is_base_of<dena::run_every_n_steps_node, T_n>::value){
426  int p = main_storage->iter_counter[node.tag];
427  main_storage->iter_counter[node.tag] = (p+1) % node.period;
428  if(p == 0)
429  (*traverse_fn)(node.sub_procedure.front(), problem, main_storage);
430  }
431  if constexpr (std::is_base_of<dena::run_until_no_improve_node, T_n>::value){
432  int checks = 0;
433  T_e value = main_storage->container(node.id)->best_min();
434  while(checks < node.max_check){
435  T_e last_value = main_storage->container(node.id)->best_min();
436  checks++;
437  if(last_value < value)
438  checks = 0;
439  value = last_value;
440  // run the sub-procedure
441  (*traverse_fn)(node.sub_procedure.front(), problem, main_storage);
442  }
443 
444  }
445  return;
446  }
447  if constexpr (std::is_base_of<dena::bcd_mask_node, T_n>::value){
448  if constexpr(T_block_dim == T_dim)
449  return;
450  // synchronize best values for the current state over the cluster
451  main_storage->update_partial_best();
452  main_storage->sync_partial_best();
453  spdlog::info("synchronizing BCD mask. best solution: {}", main_storage->partial_best->values[0]);
454  // generate a new mask
455  main_storage->str_storage[node.tag][0]->apply();
456  // synchronize the generated mask
457  main_storage->str_storage[node.tag][1]->apply();
458  // reset all solution containers
459  main_storage->reset();
460  return;
461  }
462  if (main_storage->str_storage.find(node.tag) != main_storage->str_storage.end()){
463  for(auto& str: main_storage->str_storage[node.tag]){
464  str->apply();
465  main_storage->update_partial_best();
466  }
467  }
468 
469 
470  }
471 };

477 template<typename T_e, int T_dim, int T_block_dim=T_dim>
478 class basic_runtime{
479 public:
480  // objective system
481  system<T_e>* problem;
482  std::unique_ptr<blocked_system<T_e>> blocked_problem;
483  runtime_storage<T_e, T_dim, T_block_dim> storage;
484 
485  // check if the objective system is blocked
486  constexpr bool blocked(){
487  return T_dim != T_block_dim;
488  }
489  // get problem
490  system<T_e>* get_problem(){
491  if(blocked())
492  return blocked_problem.get();
493  else
494  return problem;
495  }
496  basic_runtime(system<T_e>* problem){
497  this->problem = problem;
498  storage.partial_best = std::make_unique<basic_scontainer<T_e, T_block_dim>>(1, 1);
499  storage.partial_best->allocate();
500  if (blocked()){
501  storage.bcd_mask.resize(T_block_dim);
502  std::iota(storage.bcd_mask.begin(), storage.bcd_mask.end(), 0);
503  this->blocked_problem = std::make_unique<blocked_system<T_e>>(problem, T_dim, T_block_dim, storage.bcd_mask.data());
504  sync_bcd_mask<T_e, T_block_dim> sync_mask_str(storage.bcd_mask.data());
505  spdlog::info("broadcasting BCD mask...");
506  sync_mask_str.apply();
507  // initialize and sync the bcd state
508  storage.blocked_state = std::make_unique<basic_scontainer<T_e, T_dim>>(1, 1);
509  storage.blocked_state->allocate();
510  uniform_init_strategy<T_e, T_dim> init_bcd_state(problem, storage.blocked_state.get());
511  init_bcd_state.apply();
512  // briadcast the initialized solution
513  spdlog::info("broadcasting initial BCD solution state...");
514  sync_broadcast_best<T_e, T_dim> sync_bcd_state_str(storage.blocked_state.get());
515  sync_bcd_state_str.apply();
516  storage.th_blocked_states = tbb::enumerable_thread_specific<std::vector<T_e>>(storage.blocked_state->particles[0]);
