25template<typename T_e, int T_dim>
26class dimension_tweak_strategy: public mutation_strategy<T_e, T_dim>{
27protected:
28 // system
29 system<T_e>* problem_;
30 // main container
31 basic_scontainer<T_e, T_dim>* target_container_;
32 // candidate container
33 basic_scontainer<T_e, T_dim>* candidates_;
34 // number of mutations
35 int n_mutations_;
36 // number of affected dimensions
37 int k_;
38public:
39 dimension_tweak_strategy(system<T_e>* problem, basic_scontainer<T_e, T_dim>* tgt_container, basic_scontainer<T_e, T_dim>* cnd_container, int k){
40 this->problem_ = problem;
41 this->target_container_ = tgt_container;
42 this->candidates_ = cnd_container;
43 this->k_ = k;
44 this->n_mutations_ = cnd_container->n_particles();
45 }
46 virtual void tweak(int p_ind, int dim) = 0;
47 virtual void apply(){
48 tbb::parallel_for(0, n_mutations_, [this](int p){
49 int samples[1];
50 this->target_container_->sample_n_particles(samples, 1);
51 std::copy(this->target_container_->particle(samples[0]),
52 this->target_container_->particle(samples[0])+T_dim,
53 this->candidates_->particle(p));
54 // apply mutation on k dimensions
55 for(int d=0; d<k_; d++){
56 // choose a random dim
57 int dim = this->target_container_->sample_dim();
58 this->tweak(p, dim);
59 }
60 });
61 target_container_->replace_with(candidates_);
62 }
63};

103template<typename T_e, int T_dim>
104class multipoint_crossover: public crossover_strategy<T_e, T_dim>{
105protected:
106 // system
107 system<T_e>* problem_;
108 // main container
109 basic_scontainer<T_e, T_dim>* container_;
110 // condidates container
111 basic_scontainer<T_e, T_dim>* candidates_;
112 // number of crossovers
113 int n_crossovers_;
114 // maximum number of affected dimensions
115 int k_;
116public:
117 multipoint_crossover(system<T_e>* problem, basic_scontainer<T_e, T_dim>* container, basic_scontainer<T_e, T_dim>* candidates, int k){
118 this->k_ = k;
119 this->problem_ = problem;
120 this->container_ = container;
121 this->candidates_ = candidates;
122 this->n_crossovers_ = candidates->n_particles() / 2;
123 }
124 virtual void apply(){
125 tbb::parallel_for(0, n_crossovers_, [this](int p){
126 int parents[2];
127 this->container_->sample_n_particles(parents, 2);
128 // copy thee parents
129 for(int i=0; i<2; i++)
130 std::copy(this->container_->particle(parents[i]),
131 this->container_->particle(parents[i])+T_dim,
132 this->candidates_->particle(2*p+i));
133 // affected dims
134 std::set<int> dims;
135 for(int d=0; d<k_; d++)
136 dims.insert(this->container_->sample_dim());
137 // apply the crossover
138 for(auto dim: dims)
139 std::swap(this->candidates_->particles[parents[0]][dim],
140 this->candidates_->particles[parents[1]][dim]);
141 });
142 container_->replace_with(candidates_);
143 }
144};

147template<typename T_e, int T_dim>
148class static_segment_crossover: public crossover_strategy<T_e, T_dim>{
149protected:
150 // system
151 system<T_e>* problem_;
152 // main container
153 basic_scontainer<T_e, T_dim>* container_;
154 // main container
155 basic_scontainer<T_e, T_dim>* candidates_;
156 // length of the segment for crossover
157 int segment_length_;
158 // number of crossovers
159 int n_crossovers_;
160public:
161 static_segment_crossover(system<T_e>* problem, basic_scontainer<T_e, T_dim>* container, basic_scontainer<T_e, T_dim>* cnd_container, int segment_length){
162 this->n_crossovers_ = cnd_container->n_particles() / 2;
163 this->segment_length_ = segment_length;
164 this->problem_ = problem;
165 this->container_ = container;
166 this->candidates_ = cnd_container;
167 }
168 virtual void apply(){
169 tbb::parallel_for(0, n_crossovers_, [this](auto ci){
170 // select two distinct parents
171 int parents[2];
172 this->container_->sample_n_particles(parents, 2);
173 // select a random point for cross over
174 std::uniform_int_distribution<> point_dist(0, T_dim - segment_length_ - 1);
175 int point = point_dist(rocky::utils::random::prng());
176 // produce two cantidates
177 for(int i=0; i<2; i++){
178 // copy the solution
179 std::copy(this->container_->particle(parents[i]),
180 this->container_->particle(parents[i])+T_dim,
181 this->candidates_->particle(2*ci+i));
182 // replace the segment
183 std::copy(this->container_->particle(parents[1-i])+point,
184 this->container_->particle(parents[1-i])+point+segment_length_,
185 this->candidates_->particle(2*ci+i)+point);
186 }
187 });
188 // evaluate the candidates
189 this->candidates_->evaluate_and_update(problem_);
190 this->container_->replace_with(this->candidates_);
191 }
192};
