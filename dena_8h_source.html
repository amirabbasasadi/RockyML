158class node{
159public:
160 static std::vector<flow_node_variant> nodes;
161 static std::map<int, int> next_node;
162 template<typename T_n>
163 static int register_node(T_n node){
164 node.tag = nodes.size();
165 nodes.push_back(node);
166 next_node[node.tag] = -1;
167 return node.tag;
168 }
169 static void register_link(int s, int e){
170 next_node[s] = e;
171 }
172 static int next(int tag){
173 return next_node[tag];
174 }
175};

178class flow{
179public:
180 std::vector<int> procedure;
181 size_t total_memory;
182
183 flow& operator >>(const flow& f){
184 // concat the two procedures
185 auto last_node = procedure.back();
186 auto first_node = f.procedure.front();
187 node::register_link(last_node, first_node);
188 this->procedure.insert(this->procedure.end(), f.procedure.begin(), f.procedure.end());
189 return *this;
190 }
191};
