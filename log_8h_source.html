33 this->filename = filename;
34 this->log_output = &log_output;
35 this->step = 0;

42 void open(){
43 // create a specific filename for this rank
44 int mpi_rank;
45 MPI_Comm_rank(MPI_COMM_WORLD, &mpi_rank);
46 std::string process_spc_filename = fmt::format("proc_{}_{}", mpi_rank, filename);
47 // initialize the output file
48 log_output->open(process_spc_filename, std::fstream::out);
49 }
50 std::fstream& stream(){
51 return *log_output;
52 }
53 void write_header(){
54 if (this->log_groups)
55 this->stream() << "step,group,best" << std::endl;
56 else
57 this->stream() << "step,best" << std::endl;
58 }
59 virtual void save(){
60 this->stream() << "---" << std::endl;
61 if(this->log_output->is_open())
62 this->log_output->close();
63 }
64 virtual ~local_optimization_log(){
65 this->save();
66 }

101struct comet_optimization_log{
102 std::string comet_api_key_;
103 std::string workspace_;
104 std::string project_;
105 std::string metric_name_;
106 std::string experiment_name_;
107 std::string experiment_link_;
108 std::string experiment_key_;
109
110 comet_optimization_log(std::string comet_api_key, std::string workspace, std::string project, std::string metric_name){
111 this->comet_api_key_ = comet_api_key;
112 this->workspace_ = workspace;
113 this->project_ = project;
114 this->metric_name_ = metric_name;
115 // we need to make sure in an MPI app only the root process will create the experiment
116 this->create_experiment();
117 }
118 void create_experiment(){
119 spdlog::info("creating comet experiment...");
120 // encoding experiment data
121 nlohmann::json experiment_data = {{"workspaceName", this->workspace_},
122 {"projectName", this->project_}};
123 // sending a post request for creating the experiment
124 cpr::Response r = cpr::Post(cpr::Url{"https://www.comet.com/api/rest/v2/write/experiment/create"},
125 cpr::Header{{"Authorization", this->comet_api_key_},
126 {"Content-type", "application/json"},
127 {"Accept", "application/json"}},
128 cpr::Body{experiment_data.dump()});
129 // warn the user if connection wasn't successful
130 this->connection_warning(r.status_code);
131 nlohmann::json rdata = nlohmann::json::parse(r.text);
132 // store the exepriment information
133 this->experiment_name_ = rdata["name"];
134 this->experiment_link_ = rdata["link"];
135 // experiment key will be needed for submitting metric values
136 this->experiment_key_ = rdata["experimentKey"];
137 std::string proc_metric_name = get_metric_name();
138 spdlog::info("{} experiment name : {}", proc_metric_name, this->experiment_name_);
139 spdlog::info("{} experiment link : {}", proc_metric_name, this->experiment_link_);
140 spdlog::info("{} experiment key : {}", proc_metric_name, this->experiment_key_);
141 }
142 void broadcast_experiment(){
143 // root process should broadcast the experiment info
144 int mpi_rank;
145 MPI_Comm_rank(MPI_COMM_WORLD, &mpi_rank);
146 // a buffer for experiment key
147 char key_buffer[33];
148 if(mpi_rank == 0)
149 sprintf(key_buffer, "%.32s", experiment_key_.c_str());
150 spdlog::info("broadcasting experiment information to all processes...");
151 MPI_Bcast(key_buffer, 32, MPI_CHAR, 0, MPI_COMM_WORLD);
152 if(mpi_rank != 0){
153 experiment_key_ = std::string(key_buffer);
154 spdlog::info("process {} has the key : {}", mpi_rank, experiment_key_);
155 }
156 }
157 std::string get_metric_name(){
158 int mpi_rank;
159 MPI_Comm_rank(MPI_COMM_WORLD, &mpi_rank);
160 std::string name = fmt::format("proc_{}_{}", mpi_rank, metric_name_);
161 return name;
162 }
163 void connection_warning(int status_code){
164 if(status_code != 200)
165 spdlog::warn("Error while connecting Comet server! request status code : {}", status_code);
166 }
167};
